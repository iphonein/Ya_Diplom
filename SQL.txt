Task 1SELECT c.login, count(*) AS delivery_countFROM "Orders" as oINNER JOIN "Couriers" as c ON o."courierId"=c.idWHERE o."inDelivery"='t'GROUP BY c.login;Task 2SELECT track, CASE WHEN finished='t' THEN 2 WHEN cancelled='t' THEN -1 WHEN "inDelivery"='t' THEN 1 ELSE 0 END AS status FROM "Orders";
